Installation

Include the styles in head

<link rel="stylesheet" href="css/aren.css">

Include jQuery and the script before the closing body tag:

<script src="js/jquery-1.11.3.min.js"></script>
<script src="js/aren.min.js"></script>

Usage

Prepare the required mark-up:

<button id="aren-nav-toggle" tabindex="0" type="button" class="aren-nav-toggle aren-theme-dark" data-target="#nav" aria-controls="nav">
  <span class="icon-bar"></span>
  <span class="icon-bar"></span>
  <span class="icon-bar"></span>
</button>
<nav id="nav" role="navigation" class="aren-nav aren-theme-dark">
  <ul>
    <li><a href="#home">Home</a></li>
    <li><a href="#about">About</a>
      <ul>
        <li><a href="#people">People</a>
          <ul>
            <li><a href="#ceo">CEO</a></li>
          </ul>
        </li>
      </ul>
    </li>
    <li><a href="#link1">Contact</a></li>
  </ul>
</nav>

Initialize the nav:

<script>
    jQuery( document ).ready(function($) {
        $('#nav').aren();
    });
</script>
