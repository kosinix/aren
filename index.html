AREN

AREN - Accessible Responsive Nav

View on GitHub Download .zip Download .tar.gz

Features

  • Keyboard support
  • ARIA support
  • Responsive
  • Overridable defaults
  • Themes (or none)
  • Tested with NVDA (screen reader for the blind)

Installation

Include the styles in head

<link rel="stylesheet" href="css/aren.css">

Include jQuery and the script before the closing body tag:

<script src="js/jquery-1.11.3.min.js"></script>
<script src="js/aren.min.js"></script>

Usage

Prepare the required mark-up:

<button id="aren-nav-toggle" tabindex="0" type="button" class="aren-nav-toggle aren-theme-dark" data-target="#nav" aria-controls="nav">
    <span class="icon-bar"></span>
    <span class="icon-bar"></span>
    <span class="icon-bar"></span>
</button>
<nav id="nav" role="navigation" class="aren-nav aren-theme-dark">
    <ul>
        <li><a href="#home">Home</a></li>
        <li><a href="#about">About</a>
            <ul>
                <li><a href="#people">People</a>
                    <ul>
                        <li><a href="#ceo">CEO</a></li>
                    </ul>
                </li>
                <li><a href="#company">Company</a>
                    <ul>
                        <li><a href="#board">Board</a></li>
                        <li><a href="#history">History</a>
                            <ul>
                                <li><a href="#two">Two Years</a></li>
                                <li><a href="#three">Three Years</a></li>
                            </ul>
                        </li>
                    </ul>
                </li>
            </ul>
        </li>
        <li><a href="#contact">Contact</a></li>
    </ul>
</nav>

Initialize the nav:

<script>
    jQuery( document ).ready(function($) {
        $('#nav').aren();
    });
</script>

Overriding the Default Classes

<script>
    jQuery( document ).ready(function($) {
        $('#nav').aren({
            cssClass: {
                nav: 'aren-nav',
                expander: 'aren-expander',
                showNav: 'aren-nav-show',
                collapsed: 'aren-nav-collapsed',
                expanded: 'aren-nav-expanded',
                parent: 'aren-nav-parent',
                breakout: 'aren-breakout',
                toggle: 'aren-nav-toggle',
                toggleActive: 'aren-toggle-active',
                subMenu: 'aren-sub-menu'
            }
        });
    });
</script>

Make sure to change the class names in CSS when overriding the defaults

Overriding the Generated Markup

<script>
    jQuery( document ).ready(function($) {
        $('#nav').aren({
            markup: {
                expander: '<button class="%s" aria-haspopup="true" aria-expanded="false"></button>',
                breakout: '<div class="%s"></div>'
            }
        });
    });
</script>

Custom Event Callbacks

<script>
    jQuery( document ).ready(function($) {
        $('#nav').aren({
            breakout: [
                { // Callbacks for mobile nav
                    callback: {
                        onResize: function(css, nav){
                            // When browser is resized
                        },
                        onDocumentClick: function (e, css) {
                            // When anywhere outside of nav was clicked
                        },
                        onNavClick: function(e, css, nav){
                            // When anywhere inside of nav was clicked except expander
                        },
                        onExpanderClick: function (e, css, expander) {
                            // When expander was clicked
                        },
                        onExpanderKeyDown: function (e, css, expander) {
                            // When keyboard is pressed on expander
                        },
                        onToggleClick: function(e, css, toggle){
                            // When toggler is clicked
                        }
                    }
                } ,
                { // Callbacks for desktop
                    callback: {
                        onResize: function(css, nav){
                            // When browser is resized
                        },
                        onDocumentClick: function (e, css) {
                            // When anywhere outside of nav was clicked
                        },
                        onNavClick: function(e, css, nav){
                            // When anywhere inside of nav was clicked except expander
                        },
                        onExpanderClick: function (e, css, expander) {
                            // When expander was clicked
                        },
                        onExpanderKeyDown: function (e, css, expander) {
                            // When keyboard is pressed on expander
                        },
                        onToggleClick: function(e, css, toggle){
                            // When toggler is clicked. Might not be needed on desktop
                        }
                    }
                }
            ]
        });
    });
</script>
